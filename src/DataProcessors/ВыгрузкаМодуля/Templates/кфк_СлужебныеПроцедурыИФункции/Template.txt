// Возвращает свойство сообщения
// 
// Параметры:
//	КомпонентыОбмена - см. кфкОбменДаннымиXDTOСервер.ИнициализироватьКомпонентыОбмена 
//	ИмяСвойства - Строка - Имя свойства
// 
// Возвращаемое значение:
//	Неопределено, Произвольный - Значение свойства сообщения
Функция СвойствоСообщения(КомпонентыОбмена, ИмяСвойства)
	
	Результат = Неопределено;
	Если Не КомпонентыОбмена.СвойстваСообщения.Свойство(ИмяСвойства, Результат) Тогда
		ВызватьИсключение СтрШаблон(НСтр("ru = 'Неправильно задано имя свойства сообщения: %1'"), ИмяСвойства);
	КонецЕсли;
	
	Возврат Результат;
	
КонецФункции

// Устанавливает флаг пропуска и добавляет комментарий в протокол
// 
// Параметры:
//	КомпонентыОбмена - см. кфкОбменДаннымиXDTOСервер.ИнициализироватьКомпонентыОбмена
//	ТекстСообщения - Строка - Текст сообщения
Процедура Пропустить(КомпонентыОбмена, ТекстСообщения = "")
	
	КомпонентыОбмена.ФлагПропустить = Истина;
	
	Если Не ПустаяСтрока(ТекстСообщения) Тогда
		КомпонентыОбмена.ПротоколКонвертации.Добавить(ТекстСообщения);
	КонецЕсли;
	
КонецПроцедуры

// Добавляет ошибку в протокол.
// 
// Параметры:
//	КомпонентыОбмена - см. кфкОбменДаннымиXDTOСервер.ИнициализироватьКомпонентыОбмена 
//	ТекстОшибки - Строка - Текст ошибки
Процедура Ошибка(КомпонентыОбмена, ТекстОшибки)
	
	КомпонентыОбмена.ФлагОшибки = Истина;
	
	Если Не ПустаяСтрока(ТекстОшибки) Тогда
		КомпонентыОбмена.ПротоколКонвертации.Добавить(ТекстОшибки);
	КонецЕсли;
	
КонецПроцедуры

// Это регистрация при записи.
// 
// Параметры:
//	КомпонентыОбмена - см. кфккфкОбменДаннымиXDTOСервер.ИнициализироватьКомпонентыОбмена
// 
// Возвращаемое значение:
//  Булево - Это регистрация при записи
Функция ЭтоРегистрацияПриЗаписи(КомпонентыОбмена)
	
	Если КомпонентыОбмена.НаправлениеОбмена <> "Отправка" Тогда
		ВызватьИсключение НСтр("ru = 'Неверное направление обмена!'");
	КонецЕсли;
	
	Событие = СвойствоСообщения(КомпонентыОбмена, "СобытиеРегистрации");
	Если Событие = Неопределено Тогда
		ВызватьИсключение НСтр("ru = 'Не указано событие регистрации!'");
	КонецЕсли;
	
	Возврат Событие = Перечисления.кфкСобытияРегистрации.ПриЗаписи;
	
КонецФункции
